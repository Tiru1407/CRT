Write a Program to find the maximum number in a list.

Read elements upto -1

1<=n<=10^7

print output

-1

class Node:
    def __init__(self,x):
        self.data=x
        self.next=None
class Linkedlist:
    def __init__(self):
        self.head=self.tail=None
    def create(self,x):
        n=Node(x)
        if self.head==None:
            self.head=self.tail=n
        else:
            self.tail.next=n
            self.tail=n
    def maxi(self):
        temp=self.head
        max=0
        while temp:
            if temp.data>max:
                max=temp.data
                temp=temp.next
            else:
                temp=temp.next
        print(max)
l1=Linkedlist()
